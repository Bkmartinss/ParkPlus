CREATE DATABASE parkplus_db;

CREATE TABLE TIPO_VEICULO(
	ID INT AUTO_INCREMENT PRIMARY KEY,
	TIPO VARCHAR(50) NOT NULL
);

CREATE TABLE VEICULOS(
	ID INT AUTO_INCREMENT PRIMARY KEY,
	PLACA VARCHAR(10) NOT NULL,
	ENTRADA TIME NOT NULL,
	SAIDA TIME, 
	TEMPO INT, 
	DATA DATE,
	VALOR DECIMAL(5,2),

	VEI_TIPO INT,

	FOREIGN KEY (VEI_TIPO) REFERENCES TIPO_VEICULO(ID)
);


CREATE TABLE TARIFA(
	ID INT AUTO_INCREMENT PRIMARY KEY,
	TAR_INICIAL DECIMAL(5,2) NOT NULL,
	TAR_HORA DECIMAL(5,2) NOT NULL,

	TAR_TIPO INT,

	FOREIGN KEY (TAR_TIPO) REFERENCES TIPO_VEICULO(ID)
);
